#include <stddef.h>

size_t get_count(const char *s)
{
    char vowels[] = "aeiou";
    size_t count = 0;
    while (*s)
    {
        for (int i = 0; i < 5; i++)
        {
            if (*s == vowels[i])
            {
                count++;
                break;
            }
        }
        s++;
    }
    return count;
}

#define VOWELS "aeiou"
size_t get_count(const char *s)
{
    int count = 0;
    for (; *s != '\0'; s++)
        if (strchr(VOWELS, *s) != NULL) count++;
    return (count);
}
